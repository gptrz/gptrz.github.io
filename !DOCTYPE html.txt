<body>
  <h1>🎉 Sorteo GPTRZ 🎉</h1>
  <button onclick="sortear()">Sortear</button>
  <p id="resultado"></p>

  <script>
    let nombres = [];

    // Cargar el archivo TXT
    fetch('nombres.txt')
      .then(response => response.text())
      .then(texto => {
        nombres = texto.split('\n').map(n => n.trim()).filter(n => n.length > 0);
      });

    // Función para sortear
    function sortear() {
      if (nombres.length === 0) {
        document.getElementById('resultado').textContent = "No hay nombres cargados.";
        return;
      }
      const ganador = nombres[Math.floor(Math.random() * nombres.length)];
      document.getElementById('resultado').textContent = "El ganador es: " + ganador;
    }
  </script>
